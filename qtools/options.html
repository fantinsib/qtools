325def fetch_option_data(ticker:str, maturities: list[str], o_type: OptionType)->pd.DataFrame:
326    """ 
327    Return an option chain for a given ticker and maturities. 
328
329    Parameters
330    ----------
331    ticker : str 
332        The ticker of the stock of the underlying asset
333    maturities : list[str]
334        List of maturities in format yyyy-mm-dd
335    type : OptionType 
336        Type of the option
337
338    Returns
339    -------
340    pandas.DataFrame 
341        DataFrame containing option data
342
343    Notes
344    -----
345    All the data comes from yfinance.
346    The price of the underlying S is computed with with yfinance latest price. 
347    """
348
349    dfs = []
350    stock = yf.Ticker(ticker)
351    s = stock.fast_info["last_price"]
352    for d in maturities:
353        if o_type == OptionType.EUROPEAN_CALL:
354            option = stock.option_chain(d).calls
355        if o_type == OptionType.EUROPEAN_PUT:
356            option = stock.option_chain(d).puts
357        option["maturity"] = d
358        dfs.append(option)
359
360    df = pd.concat(dfs, ignore_index=True)
361    if df.empty:
362        raise FetchedDataError("options.fetch_option_data : could not retrieve any data from source.")
363    if df.isna().sum().sum() > 0: 
364        warnings.warn("options.fetch_option_data : fetched data contains nan values.", FetchedDataWarning)
365    df["type"] = o_type
366    df["S"] = s
367    return df

369def to_quote(df: pd.DataFrame, r: float, valuation_date: Optional[str] = None, ticker: Optional[str]= None) -> list[OptionQuote]:
370    """
371    Converts a fetched df from fetch_option_data to a list of OptionQuote
372
373    Parameters
374    ----------
375    df : pandas.DataFrame
376        The dataframe containing the option chain
377    r : float 
378        The risk-free rate
379    valuation_date : Optional[str]
380        Date used to compute time to maturity in years; default will use today's date
381    ticker : Optional[str]
382        Ticker of the underlying asset
383
384    Returns
385    -------
386    list[OptionQuote]
387        A list of OptionQuote extracted from the df.
388    """
389    
390    quote_list = []
391    for _, row in df.iterrows():
392        T = time_to_mat(row.maturity, valuation_date)
393        contract = OptionContract(row.type, maturity=row.maturity, T=T, K=row.strike, ticker=ticker)
394        market = MarketState(row.S, r)
395        quote_list.append(OptionQuote(contract, market, market_price = (row.bid+row.ask)/2, bid=row.bid, ask=row.ask)) #compute mid price
396
397    return quote_list

399def compute_T(quote : OptionQuote, valuation_date: Optional[str] = None):
400    """
401    Returns a new OptionQuote with T computed.
402
403    Parameters
404    ----------
405    quote : OptionQuote
406        The OptionQuote on which to compute T
407    valuation_date : Optional[str]
408        Date from which to compute. Default will use today's date
409
410    Returns

412    OptionQuote

421def time_to_mat( maturity: str, today: Optional[str] = None)->float:
422    """
423    Returns the time to maturity in years
424    
425    Parameters
426    ----------
427    maturity : str
428        The date of expiry of the contract.
429    today : Optional[str]
430        Date from which to compute time to maturity. Default will compute from today's date
431
432    Returns
433    -------
434    float
435        The time to maturity in years.
436    """
437    expiry = datetime.strptime(maturity, "%Y-%m-%d")
438
439    if not today:
440        now = datetime.now()
441    else:
442        now = datetime.strptime(today, "%Y-%m-%d")
443    return(expiry - now).total_seconds() / (365.0 * 24.0 * 3600.0)
